plot(survfit(fit), lty = 2:3, xlab = "time", ylab = "Survival")
legend("topright",legend=c("female","male"), lty = 2:3)

plot(survfit(fit),fun="cumhaz", lty = 1:2, xlab = "time", ylab = "Cumhaz")
